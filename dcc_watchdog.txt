var audio = new Audio('https://thezapalsky.github.io/tindeck_1.mp3');
var audio2 = new Audio('https://thezapalsky.github.io/tindeck_1.mp3/beep.mp3');
var audio3 = new Audio('https://thezapalsky.github.io/tindeck_1.mp3/bleep.mp3');
var notification_status = true;
var square = document.createElement('div');

square.id='sq1';
square.style.width="25px";
square.style.height="25px";
square.style.backgroundColor ='green';
square.style.position="absolute";
square.style.top='0px'
square.style.zIndex='100'
document.body.appendChild(square);
var sq1 = document.getElementById('sq1');
document.addEventListener('keypress', logKey);
function logKey(e) {
	if(e.keyCode == 42){
		notification_status=!notification_status;
		if(notification_status){
			audio2.play();
			sq1.style.backgroundColor='green';
		}
		else{
			sq1.style.backgroundColor='red';
			audio3.play();
		}
	}
	console.log(e.keyCode);
}
setInterval(function(){
    if(notification_status){
 		if(document.getElementById('row0eventGrid').children[10].children[0].title == 'Open'){
			audio.play();
			console.log('beep');
        }
	console.log('monitoring...');
    }
	
}, 10000);
